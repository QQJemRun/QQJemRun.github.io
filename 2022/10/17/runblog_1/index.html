聊一下QT自带静态数据库sqlite实现登陆检索的粗略思路

首先一定是sql基本语法方面(教程很多，只说用到的)。

//在数据库中创建一个表格
//integer primary key 可以实现数据的自增，经常是表格第一列
create table client(id integer primary key,name_1 type,name_2 type);
//插入语法
insert into database_name.table_name(colume1,colume2...) values(values1,values2);//插入语法
//检索数据*_1可以是表格的列名，也可以不指定直接用\*表示，*_2是表格的名字，where可以添加限制条件
select *_1 from *_2 where *_3;

//以上语法为sqlite语法，后头的“；”是必须的（头铁的撞）

有了基本语法就可以实现一些操作了

1.Qt对数据库的操作可以使用QSqlQuery类来实现，在这里我只用了QSqlQuery::exec()方法。

2.具体操作时还可以借助QSqlQuery::prepare()方法来确保程序的健壮性。

3.Qt中数据库文件用QSqlDatabase类来标识，这里只用到QSqlDatabase::open()方法和QSqlDatabase::close()方法

4.数据库文件用完记得关闭！！！

5.做项目删除数据时备份成为静态文件是一个良好的修养(正经脸)。

#ifndef WIDGET_H

class Widget : public QWidget
{
    Q_OBJECT

public:
    Widget(QWidget *parent = nullptr);
    ~Widget();

private slots:
    void on_pushButton_clicked();

    void on_pushButton_2_clicked();

private:
    Ui::Widget *ui;
    QSqlDatabase db;
    QSqlQuery query;
    mywidget_1 mainwidget;
};
#endif // WIDGET_H

.cpp文件内容：

<?cpp
#include "mywidget.h"
#include "ui_mywidget.h"
#include <QMovie>
#include <QDebug>
#include <QMessageBox>
#include <QSqlQuery>
#include <QSqlError>
#include <QString>
#include "mywidget_1.h"

Widget::Widget(QWidget *parent)
    : QWidget(parent)
    , ui(new Ui::Widget)
{
    ui->setupUi(this);
    //设置头像动画
    QMovie *mov = new QMovie(":/q.gif");
    mov->setScaledSize(QSize(ui->mylabel->width(),ui->mylabel->height()));
    mov->start();
    ui->mylabel->setMovie(mov);
    //查看数据库驱动：qDebug()<<QSqlDatabase::drivers();
    //建立和SQlite数据库的连接
    db = QSqlDatabase::addDatabase("QSQLITE");
    //设置数据库文件的路径和名字
    db.setDatabaseName("./run.db");
    //创建数据库表格
    if(!db.open())
    {
        QMessageBox i = QMessageBox(this);
        i.warning(this,"sss","ssss");
        return;
    }
    query = QSqlQuery(db);
    //sqlite创建表的
    query.exec("create table client(id integer primary key,name text not null,password text not null);");
    db.close();
}

Widget::~Widget()
{
    delete ui;
}

//注册按钮
void Widget::on_pushButton_clicked()
{
    //用户未输入
    if(ui->lineEdit->text().isEmpty() || ui->lineEdit_2->text().isEmpty())
    {
        QMessageBox i = QMessageBox(this);
        i.warning(this,"失败：","请输入用户名和密码！！！");
        return;
    }
    if(!db.open())
    {
        QMessageBox i = QMessageBox(this);
        i.warning(this,"错误：","打开数据库失败！！！");
        return;
    }
    //注册验证
    if(!query.exec(QString("select * from client where name='%1';").arg(ui->lineEdit->text())))
    {
        qDebug() << query.lastError();
    }
    else
    {
        while(query.next())
        {
            //核对用户名
            if(query.value(1).toString() == ui->lineEdit->text())
            {
                QMessageBox i = QMessageBox(this);
                i.warning(this,"失败：","用户已经存在！！！");
                return;
            }
        }
        //添加用户帐号
        //sqlite插入语法:insert into database_name.table_name(colume1,colume2...) values(values1,values2);
        query.prepare("insert into client(name,password) values ( ?,? );");
        query.addBindValue(ui->lineEdit->text());
        query.addBindValue(ui->lineEdit_2->text());
        if(!query.exec())
        {
            qDebug() << query.lastError();
        }
    }
    db.close();
}

//登陆按钮
void Widget::on_pushButton_2_clicked()
{
    //用户未输入
    if(ui->lineEdit->text().isEmpty() || ui->lineEdit_2->text().isEmpty())
    {
        QMessageBox i = QMessageBox(this);
        i.warning(this,"失败：","请输入用户名和密码！！！");
        return;
    }
    if(!db.open())
    {
        QMessageBox i = QMessageBox(this);
        i.warning(this,"错误：","打开数据库失败！！！");
        return;
    }

    if(!query.exec(QString("select * from client where name = '%1';").arg(ui->lineEdit->text())))
    {
        qDebug() <<query.lastError();
    }
    else
    {
        //数据库中存在该用户
        while(query.next())
        {
            //核对密码
            if(ui->lineEdit_2->text() != query.value(2))
            {
                QMessageBox i = QMessageBox(this);
                i.warning(this,"失败：","密码错误！！！");
                return;
            }
            //qDebug() << QString("%1  ,  %2  ,  %3").arg(query.value(0).toInt()).arg(query.value(1).toString()).arg(query.value(2).toString() );
            //this->hide();
            this->mainwidget.show();//进入新的窗口。
        }
        //未从数据库查询到用户信息
        if(!query.first())
        {
            QMessageBox i = QMessageBox(this);
            i.warning(this,"失败：","用户名不存在！！！");
            return;
        }
    }
    db.close();
}
